Also create helper for cookie setting to avoid duplication:
```go
func (h *Handler) setAuthCookies(w http.ResponseWriter, userID, email string) error {
    accessToken, err := h.jwt.CreateAccessToken(userID, email)
    if err != nil { return err }
    refreshToken, err := h.jwt.CreateRefreshToken(userID, email)
    if err != nil { return err }
    http.SetCookie(w, &http.Cookie{
        Name: "access_token", Value: accessToken, Path: "/",
        HttpOnly: true, SameSite: http.SameSiteLaxMode, MaxAge: 900,
    })
    http.SetCookie(w, &http.Cookie{
        Name: "refresh_token", Value: refreshToken, Path: "/api/auth",
        HttpOnly: true, SameSite: http.SameSiteLaxMode, MaxAge: 604800,
    })
    return nil
}
```
